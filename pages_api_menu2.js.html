/**
 * @fileoverview API endpoint for retrieving menu items with favorite status calculation
 * @author Dhiraj Atmakuri
 */

import { pool } from '@/lib/db';

/**
 * Handles GET requests for menu items with favorite status
 * @async
 * @function handler
 * @param {Object} req - The HTTP request object
 * @param {Object} res - The HTTP response object
 * @returns {Promise<void>} JSON response with menu items and their favorite status
 * @throws {Error} Database or server errors
 * 
 * @description
 * - Retrieves all menu items with their details and calculated favorite status
 * - Favorite status is determined by item type and rating thresholds:
 *   - Entrees: 4.0+ rating
 *   - Premium Entrees: 4.5+ rating
 *   - Sides: 4.2+ rating
 *   - Appetizers/Desserts/Drinks: 4.3+ rating
 */
export default async function handler(req, res) {
  if (req.method === 'GET') {
    try {
      const query = `
        SELECT 
          menu_item_id,
          price,
          name,
          amount_available,
          type,
          CASE
            WHEN type = 'entree' AND ratings_count >= 1 AND (ratings_sum::float / ratings_count) >= 4.0 THEN true
            WHEN type = 'premium entree' AND ratings_count >= 1 AND (ratings_sum::float / ratings_count) >= 4.5 THEN true
            WHEN type = 'side' AND ratings_count >= 1 AND (ratings_sum::float / ratings_count) >= 4.2 THEN true
            WHEN type IN ('appetizer', 'dessert', 'drink') AND ratings_count >= 1 AND (ratings_sum::float / ratings_count) >= 4.3 THEN true
            ELSE false
          END as is_favorite,
        FROM menu_item
        ORDER BY 
          CASE 
            WHEN type IN ('appetizer', 'dessert', 'drink') THEN 'Extra'
            WHEN type = 'premium entree' THEN 'entree'
            ELSE type 
          END,
          name;
      `;
      
      const { rows } = await pool.query(query);
      
      // Log the first few items to check the data
      console.log('Sample menu items:', rows.slice(0, 3));
      
      res.status(200).json(rows);
    } catch (error) {
      console.error('Error fetching menu items:', error);
      res.status(500).json({ error: 'Failed to fetch menu items' });
    }
  } else {
    res.setHeader('Allow', ['GET']);
    res.status(405).end(`Method ${req.method} Not Allowed`);
  }
}

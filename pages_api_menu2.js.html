/**
 * API route for retrieving menu items from the database
 * 
 * @file menu2.js
 * @module MenuItemsAPI
 * @author Dhiraj Atakumari
 * @description Handles GET requests to fetch all menu items, sorted by type and name
 * @requires @/lib/db
 */

import { pool } from '@/lib/db';

/**
 * API route handler for menu items
 * 
 * @function handler
 * @param {Object} req - Next.js API route request object
 * @param {Object} res - Next.js API route response object
 * @returns {Promise<void>}
 * @throws {Error} Throws an error if database query fails
 */
export default async function handler(req, res) {
  // Handle GET request to retrieve menu items
  if (req.method === 'GET') {
    try {
      // SQL query to fetch menu items, sorted by type and name
      const query = `
        SELECT 
          menu_item_id,
          price,
          name,
          amount_available,
          type,
          COALESCE(description, '') AS description
        FROM menu_item
        ORDER BY type, name;
      `;
      
      // Execute database query
      const { rows } = await pool.query(query);
      
      // Return menu items with 200 OK status
      res.status(200).json(rows);
    } catch (error) {
      // Log and handle any database query errors
      console.error('Error fetching menu items:', error);
      res.status(500).json({ error: 'Failed to fetch menu items' });
    }
  } else {
    // Handle unsupported HTTP methods
    res.setHeader('Allow', ['GET']);
    res.status(405).end(`Method ${req.method} Not Allowed`);
  }
}
